# 像使用 OpenAI 一样使用 UModelVerse
client = OpenAI(
    api_key="sk-UModelVerse-xxxxxxxx", 
    base_url="https://api.umodelverse.com/v1"
)

# 极速调用 Qwen 或 DeepSeek
response = client.chat.completions.create(
    model="qwen-2.5-72b-instruct",
    messages=[
        {"role": "user", "content": "介绍一下优刻得"}
    ],
    stream=True
)

for chunk in response:
    print(chunk.choices[0].delta.content or "", end="")
